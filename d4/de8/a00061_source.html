93  AddPdf(PDF1 const& pdf1, PDF2 const& pdf2, PDFs const& ...pdfs,
94  std::array<Parameter*, npdfs-1>const& coef):
95  fPDFs(thrust::make_tuple(pdf1,pdf2,pdfs...) ),
96  fExtended(kFalse),
97  fFractioned(kTrue),
98  fCoefSum(0.0)
99  {
100  size_t i=0;
101  for(Parameter* var:coef)
102  {
103  if(var->GetLowerLim()< 0.0 || var->GetUpperLim() > 1.0 ||
104  var->GetValue()< 0.0 || var->GetValue() > 1.0)
105  {
106  HYDRA_LOG(WARNING,"Fraction out of bounds" )
107  HYDRA_MSG << (*var) << HYDRA_ENDL;
108  HYDRA_MSG << "Change the parameter limits to [0.0, 1.0] and value accordingly."<< HYDRA_ENDL;
109  abort();
110  }
111 
112  fCoeficients[i]= *var;
113  fCoefSum+= var->GetValue();
114  i++;
115  }
116 
117  fCoeficients[npdfs-1]= 1.0 - fCoefSum;
118  }
