73  DecayMother(Vector4R const& mother,
74  vector_real const& _masses,
75  const GInt_t _ndaughters, const GInt_t _seed):
76  //fMasses(thrust::raw_pointer_cast(_masses.data())),
77  fNDaughters(_ndaughters),
78  fSeed(_seed)

80  {
81 
82  for(size_t i=0; i<N; i++) fMasses[i]=_masses[i];
83 
84  GReal_t _fTeCmTm = mother.mass(); // total energy in C.M. minus the sum of the masses
85 
86  for (size_t n = 0; n < fNDaughters; n++)
87  {
88  _fTeCmTm -= _masses.data()[n];
89  }
90 
91  GReal_t emmax = _fTeCmTm + _masses.data()[0];
92  GReal_t emmin = 0.0;
93  GReal_t wtmax = 1.0;
94  for (size_t n = 1; n < fNDaughters; n++)
95  {
96  emmin += _masses.data()[n - 1];
97  emmax += _masses.data()[n];
98  wtmax *= pdk(emmax, emmin, _masses.data()[n]);
99  }
100  GReal_t _fWtMax = 1.0 / wtmax;
101 
102  GReal_t _beta = mother.d3mag() / mother.get(0);
103 
104  if (_beta)
105  {
106  GReal_t w = _beta / mother.d3mag();
107  fBeta0 = mother.get(0) * w;
108  fBeta1 = mother.get(1) * w;
109  fBeta2 = mother.get(2) * w;
110  }
111  else
112  fBeta0 = fBeta1 = fBeta2 = 0.0;
113 
114  fTeCmTm = _fTeCmTm;
115  fWtMax = _fWtMax;

133  __host__ __device__ inline
134  GReal_t pdk(const GReal_t a, const GReal_t b,
135  const GReal_t c) const
136  {
137  //the PDK function
138  GReal_t x = (a - b - c) * (a + b + c) * (a - b + c) * (a + b - c);
139  x = sqrt(x) / (2 * a);
140  return x;
141  }

143  __host__ __device__ inline
144  void bbsort( GReal_t *array, GInt_t n)
145  {
146  // Improved bubble sort
147 
148  for (GInt_t c = 0; c < n; c++)
149  {
150  GInt_t nswap = 0;
151 
152  for (GInt_t d = 0; d < n - c - 1; d++)
153  {
154  if (array[d] > array[d + 1]) /* For decreasing order use < */
155  {
156  GReal_t swap = array[d];
157  array[d] = array[d + 1];
158  array[d + 1] = swap;
159  nswap++;
160  }
161  }
162  if (nswap == 0)
163  break;
164  }
165 
166  }

169  __host__ __device__ inline
170  size_t hash(size_t a, size_t b)
171  {
172  //Matthew Szudzik pairing
173  //http://szudzik.com/ElegantPairing.pdf
174 
175  size_t A = 2 * a ;
176  size_t B = 2 * b ;
177  size_t C = ((A >= B ? A * A + A + B : A + B * B) / 2);
178  return C ;

181  __host__ __device__ inline
182  GReal_t process(const GInt_t evt, Vector4R** daugters)
183  {
184 
185  GRND randEng( hash(evt,fSeed) );
186  thrust::uniform_real_distribution<GReal_t> uniDist(0.0, 1.0);
187 
188  GReal_t rno[N];
189  rno[0] = 0.0;
190  rno[fNDaughters - 1] = 1.0;
191 
192  if (fNDaughters > 2)
193  {
194  #pragma unroll N
195  for (GInt_t n = 1; n < fNDaughters - 1; n++)
196  {
197  rno[n] = uniDist(randEng) ;
198 
199  }
200 
201  bbsort(&rno[1], fNDaughters -2);
202 
203  }
204 
205 
206  GReal_t invMas[kMAXP], sum = 0.0;
207 
208  #pragma unroll N
209  for (size_t n = 0; n < fNDaughters; n++)
210  {
211  //printf("%d mass=%f \n",n, fMasses[n]);
212  sum += fMasses[n];
213  invMas[n] = rno[n] * fTeCmTm + sum;
214  }
215 
216  //
217  //-----> compute the weight of the current event
218  //
219 
220  GReal_t wt = fWtMax;
221 
222  GReal_t pd[N];
223 
224  #pragma unroll N
225  for (size_t n = 0; n < fNDaughters - 1; n++)
226  {
227  pd[n] = pdk(invMas[n + 1], invMas[n], fMasses[n + 1]);
228  wt *= pd[n];
229  }
230 
231  //
232  //-----> complete specification of event (Raubold-Lynch method)
233  //
234 
235  daugters[0]->set(sqrt((GReal_t) pd[0] * pd[0] + fMasses[0] * fMasses[0]), 0.0,
236  pd[0], 0.0);
237 
238  #pragma unroll N
239  for (size_t i = 1; i < fNDaughters; i++)
240  {
241 
242  daugters[i]->set(
243  sqrt(pd[i - 1] * pd[i - 1] + fMasses[i] * fMasses[i]), 0.0,
244  -pd[i - 1], 0.0);
245 
246  GReal_t cZ = 2 * uniDist(randEng) -1 ;
247  GReal_t sZ = sqrt(1 - cZ * cZ);
248  GReal_t angY = 2 * PI* uniDist(randEng);
249  GReal_t cY = cos(angY);
250  GReal_t sY = sin(angY);
251  for (size_t j = 0; j <= i; j++)
252  {
253 
254  GReal_t x = daugters[j]->get(1);
255  GReal_t y = daugters[j]->get(2);
256  daugters[j]->set(1, cZ * x - sZ * y);
257  daugters[j]->set(2, sZ * x + cZ * y); // rotation around Z
258 
259  x = daugters[j]->get(1);
260  GReal_t z = daugters[j]->get(3);
261  daugters[j]->set(1, cY * x - sY * z);
262  daugters[j]->set(3, sY * x + cY * z); // rotation around Y
263  }
264 
265  if (i == (fNDaughters - 1))
266  break;
267 
268  GReal_t beta = pd[i] / sqrt(pd[i] * pd[i] + invMas[i] * invMas[i]);
269  for (size_t j = 0; j <= i; j++)
270  {
271 
272  daugters[j]->applyBoostTo(Vector3R(0, beta, 0));
273  }
274 
275  }
276 
277  //
278  //---> final boost of all particles to the mother's frame
279  //
280  #pragma unroll N
281  for (size_t n = 0; n < fNDaughters; n++)
282  {
283 
284  daugters[n]->applyBoostTo(Vector3R(fBeta0, fBeta1, fBeta2));
285 
286  }
287 
288  //
289  //---> return the weight of event
290  //
291 
292  return wt;
293 
294  }
