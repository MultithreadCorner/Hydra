80  void AddParameter( hydra::Parameter* param )

113  void UpdateParameterLimits(ROOT::Minuit2::MinosError const& minos_error )
114  {
115 
116  std::string name = minos_error.LowerState().Name( minos_error.Parameter());
117  auto parameter = std::find_if(fVariables.begin(), fVariables.end(),
118  [&](hydra::Parameter* const p) { return std::string(p->GetName()) == name; } );
119 
120  if (parameter == std::end(fVariables)) {
121  HYDRA_LOG(WARNING, " Parameter :"<< name << " not found. Limits not set.\n\n")
122  return;
123  }
124  (*parameter)->SetLowerLim(minos_error.Lower() );
125  (*parameter)->SetUpperLim(minos_error.Lower() );
126 
127  return;
