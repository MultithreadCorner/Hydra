380 inline void Vector4R::applyBoostTo(const Vector3R& boost, bool inverse)
381 {
382 
383  GReal_t bx, by, bz, gamma, b2;
384 
385  bx = boost.get(0);
386  by = boost.get(1);
387  bz = boost.get(2);
388 
389  GReal_t bxx = bx * bx;
390  GReal_t byy = by * by;
391  GReal_t bzz = bz * bz;
392 
393  b2 = bxx + byy + bzz;
394 
395  if (b2 > 0.0 && b2 < 1.0)
396  {
397 
398  gamma = 1.0 / sqrt(1.0 - b2);
399 
400  GReal_t gb2 = (gamma - 1.0) / b2;
401 
402  GReal_t gb2xy = gb2 * bx * by;
403  GReal_t gb2xz = gb2 * bx * bz;
404  GReal_t gb2yz = gb2 * by * bz;
405 
406  GReal_t gbx = gamma * bx;
407  GReal_t gby = gamma * by;
408  GReal_t gbz = gamma * bz;
409 
410  GReal_t e2 = v[0];
411  GReal_t px2 = v[1];
412  GReal_t py2 = v[2];
413  GReal_t pz2 = v[3];
414 
415  if (inverse)
416  {
417  v[0] = gamma * e2 - gbx * px2 - gby * py2 - gbz * pz2;
418 
419  v[1] = -gbx * e2 + gb2 * bxx * px2 + px2 + gb2xy * py2
420  + gb2xz * pz2;
421 
422  v[2] = -gby * e2 + gb2 * byy * py2 + py2 + gb2xy * px2
423  + gb2yz * pz2;
424 
425  v[3] = -gbz * e2 + gb2 * bzz * pz2 + pz2 + gb2yz * py2
426  + gb2xz * px2;
427  }
428  else
429  {
430  v[0] = gamma * e2 + gbx * px2 + gby * py2 + gbz * pz2;
431
