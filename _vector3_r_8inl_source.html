115 inline void Vector3R::applyRotateEuler(GReal_t phi, GReal_t theta, GReal_t ksi)
116 {
117 
118  GReal_t temp[3];
119  GReal_t sp, st, sk, cp, ct, ck;
120 
121  sp = sin(phi);
122  st = sin(theta);
123  sk = sin(ksi);
124  cp = cos(phi);
125  ct = cos(theta);
126  ck = cos(ksi);
127 
128  temp[0] = (ck * ct * cp - sk * sp) * v[0] + (-sk * ct * cp - ck * sp) * v[1]
129  + st * cp * v[2];
130  temp[1] = (ck * ct * sp + sk * cp) * v[0] + (-sk * ct * sp + ck * cp) * v[1]
131  + st * sp * v[2];
132  temp[2] = -ck * st * v[0] + sk * st * v[1] + ct * v[2];
133 
134  v[0] = temp[0];
135  v[1] = temp[1];
136  v[2] = temp[2];
137 }
