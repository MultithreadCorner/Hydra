94 // ProcessCallsPlainBinary is a functor that accepts two PlainState
95 // structs and returns a new summary_stats_data which are an
96 // approximation to the summary_stats for
97 // all values that have been agregated so far
98 struct ProcessCallsPlainBinary
99  : public thrust::binary_function<PlainState const&,
100  PlainState const&,
101  PlainState >
102 {
103  __host__ __device__ inline
104  PlainState operator()(const PlainState& x, const PlainState& y)
105  {
106  PlainState result;
107 
108  // precompute some common subexpressions
109  size_t n = x.fN + y.fN;
110 
111  GReal_t delta = y.fMean - x.fMean;
112  GReal_t delta2 = delta * delta;
113 
114  //Basic number of samples (n), min, and max
115  result.fN = n;
116  result.fMin = thrust::min(x.fMin, y.fMin);
117  result.fMax = thrust::max(x.fMax, y.fMax);
118 
119  result.fMean = x.fMean + delta * y.fN / n;
120 
121  result.fM2 = x.fM2 + y.fM2;
122  result.fM2 += delta2 * x.fN * y.fN / n;
123 
124  return result;
125  }
126 };
