74 struct EvalOnDaughtersBinary
75  :public HYDRA_EXTERNAL_NS::thrust::binary_function< ResultPHSP const&, ResultPHSP const&, ResultPHSP >

79  __host__ __device__ inline
80  ResultPHSP operator()(ResultPHSP const& x, ResultPHSP const& y)
81  {
82  ResultPHSP result;
83 
84  GReal_t w = x.fW + y.fW;
85 
86  GReal_t delta = y.fMean*y.fW - x.fMean*x.fW;
87  GReal_t delta2 = delta * delta;
88 
89  result.fW = w;
90 
91  result.fMean = (x.fMean*x.fW + y.fMean*y.fW)/w;
92  result.fM2 = x.fM2 + y.fM2;
93  result.fM2 += delta2 * x.fW * y.fW / w;
94 
95  return result;
96  }

101 struct EvalOnDaughters
102 {
103 
104  const GInt_t fSeed;
105 
106  GReal_t fTeCmTm;
107  GReal_t fWtMax;
108  GReal_t fBeta0;

115 
116  //constructor
117  EvalOnDaughters(FUNCTOR const& functor, Vector4R const& mother,
118  const GReal_t (&masses)[N],
119  const GInt_t _seed):
120  fSeed(_seed),
121  fFunctor(functor)
122 
123  {
124 
125  for(size_t i=0; i<N; i++) fMasses[i]=masses[i];
126 
127  GReal_t _fTeCmTm = mother.mass(); // total energy in C.M. minus the sum of the masses
128 
129  for (size_t n = 0; n < N; n++)
130  {
131  _fTeCmTm -= masses[n];
132  }
133 
134  GReal_t emmax = _fTeCmTm + masses[0];
135  GReal_t emmin = 0.0;
136  GReal_t wtmax = 1.0;
137  for (size_t n = 1; n < N; n++)
138  {
139  emmin += masses[n - 1];
140  emmax += masses[n];
141  wtmax *= pdk(emmax, emmin, masses[n]);
142  }
143  GReal_t _fWtMax = 1.0 / wtmax;
144 
145  GReal_t _beta = mother.d3mag() / mother.get(0);
146 
147  if (_beta)
148  {
149  GReal_t w = _beta / mother.d3mag();
150  fBeta0 = mother.get(0) * w;
151  fBeta1 = mother.get(1) * w;
152  fBeta2 = mother.get(2) * w;
153  }
154  else
155  fBeta0 = fBeta1 = fBeta2 = 0.0;
156 
157  fTeCmTm = _fTeCmTm;
158  fWtMax = _fWtMax;
159 
160 
161  }

176  __host__ __device__ inline
177  static GReal_t pdk(const GReal_t a, const GReal_t b,
178  const GReal_t c)
179  {
180  //the PDK function
181  //GReal_t x = (a - b - c) * (a + b + c) * (a - b + c) * (a + b - c);
182  //x = sqrt( x ) / (2 * a);
183  return sqrt( (a - b - c) * (a + b + c) * (a - b + c) * (a + b - c) ) / (2 * a);
184  }

186  __host__ __device__ inline
187  void bbsort( GReal_t *array, GInt_t n)
188  {
189  // Improved bubble sort
190 
191  for (GInt_t c = 0; c < n; c++)
192  {
193  GInt_t nswap = 0;
194 
195  for (GInt_t d = 0; d < n - c - 1; d++)
196  {
197  if (array[d] > array[d + 1]) /* For decreasing order use < */
198  {
199  GReal_t swap = array[d];
200  array[d] = array[d + 1];
201  array[d + 1] = swap;
202  nswap++;
203  }
204  }
205  if (nswap == 0)
206  break;
207  }

212  __host__ __device__ inline
213  constexpr static size_t hash(const size_t a, const size_t b)
214  {
215  //Matthew Szudzik pairing
216  //http://szudzik.com/ElegantPairing.pdf
217  return (((2 * a) >= (2 * b) ? (2 * a) * (2 * a) + (2 * a) + (2 * b) : (2 * a) + (2 * b) * (2 * b)) / 2);

220  __host__ __device__ inline
221  GReal_t process(const GInt_t evt, Vector4R (&daugters)[N])
222  {
223 
224  GRND randEng( hash(evt,fSeed) );
225  HYDRA_EXTERNAL_NS::thrust::uniform_real_distribution<GReal_t> uniDist(0.0, 1.0);
226 
227  GReal_t rno[N];
228  rno[0] = 0.0;
229  rno[N - 1] = 1.0;
230 
231  if (N > 2)
232  {
233 #pragma unroll N
234  for (GInt_t n = 1; n < N - 1; n++)
235  {
236  rno[n] = uniDist(randEng) ;
237 
238  }
239 
240  bbsort(&rno[1], N -2);
241 
242  }
243 
244 
245  GReal_t invMas[N], sum = 0.0;
246 
247 #pragma unroll N
248  for (size_t n = 0; n < N; n++)
249  {
250  //printf("%d mass=%f \n",n, fMasses[n]);
251  sum += fMasses[n];
252  invMas[n] = rno[n] * fTeCmTm + sum;
253  }
254 
255  //
256  //-----> compute the weight of the current event
257  //
258 
259  GReal_t wt = fWtMax;
260 
261  GReal_t pd[N];
262 
263 #pragma unroll N
264  for (size_t n = 0; n < N - 1; n++)
265  {
266  pd[n] = pdk(invMas[n + 1], invMas[n], fMasses[n + 1]);
267  wt *= pd[n];
268  }
269 
270  //
271  //-----> complete specification of event (Raubold-Lynch method)
272  //
273 
274  daugters[0].set(sqrt((GReal_t) pd[0] * pd[0] + fMasses[0] * fMasses[0]), 0.0,
275  pd[0], 0.0);
276 
277 #pragma unroll N
278  for (size_t i = 1; i < N; i++)
279  {
280 
281  daugters[i].set(
282  sqrt(pd[i - 1] * pd[i - 1] + fMasses[i] * fMasses[i]), 0.0,
283  -pd[i - 1], 0.0);
284 
285  GReal_t cZ = 2 * uniDist(randEng) -1 ;
286  GReal_t sZ = sqrt(1 - cZ * cZ);
287  GReal_t angY = 2 * PI* uniDist(randEng);
288  GReal_t cY = cos(angY);
289  GReal_t sY = sin(angY);
290  for (size_t j = 0; j <= i; j++)
291  {
292 
293  GReal_t x = daugters[j].get(1);
294  GReal_t y = daugters[j].get(2);
295  daugters[j].set(1, cZ * x - sZ * y);
296  daugters[j].set(2, sZ * x + cZ * y); // rotation around Z
297 
298  x = daugters[j].get(1);
299  GReal_t z = daugters[j].get(3);
300  daugters[j].set(1, cY * x - sY * z);
301  daugters[j].set(3, sY * x + cY * z); // rotation around Y
302  }
303 
304  if (i == (N - 1))
305  break;
306 
307  GReal_t beta = pd[i] / sqrt(pd[i] * pd[i] + invMas[i] * invMas[i]);
308  for (size_t j = 0; j <= i; j++)
309  {
310 
311  daugters[j].applyBoostTo(Vector3R(0, beta, 0));
312  }
313 
314  }
315 
316  //
317  //---> final boost of all particles to the mother's frame
318  //
319 #pragma unroll N
320  for (size_t n = 0; n < N; n++)
321  {
322 
323  daugters[n].applyBoostTo(Vector3R(fBeta0, fBeta1, fBeta2));
324 
325  }
326 
327  //
328  //---> return the weight of event
329  //
330 
331  return wt;
332 
333  }
