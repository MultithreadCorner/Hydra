56 GInt_t Vegas< N, GRND >::Integrate(FUNCTOR const& fFunctor, GBool_t reset) {
57 
58  if(reset){
59  fState->SetStage(0);
60  fState->ResetState();
61  }
62 
63  GReal_t cum_int, cum_sig;
64  //cout <<"stage " <<fState->GetStage() << endl;
65 
66 
67  if( fState->GetStage() == 0) {
68  InitGrid();
69 
70  if (fState->GetVerbose() >= 0) {
71  PrintLimits();
72  }
73  }
74 
75  if( fState->GetStage() <= 1) {
76 
77  fState->SetWeightedIntSum(0.0);
78  fState->SetSumOfWeights(0.0);
79  fState->SetChiSum(0.0);
80  fState->SetItNum(1);
81  fState->SetSamples(0.0);
82  fState->SetChiSquare(0.0);
83  }
84 
85  if( fState->GetStage() <= 2) {
86 
87  size_t bins = fState->GetNBinsMax();
88  size_t boxes = 1;
89 
90  if (fState->GetMode() != MODE_IMPORTANCE_ONLY) {
91  /* shooting for 2 calls/box */
92 
93  boxes = floor( pow(fNCalls / 2.0, 1.0 / N ));
94  //if(boxes==1) boxes++;
95  fState->SetMode(MODE_IMPORTANCE);
96 
97  if (2 * boxes >= fState->GetNBinsMax() ) {
98  /* if bins/box < 2 */
99  GInt_t box_per_bin = std::max(GInt_t(boxes/fState->GetNBinsMax()), 1);
100 
101  bins = std::min(GInt_t(boxes / box_per_bin), GInt_t(fState->GetNBinsMax()));
102  boxes = box_per_bin * bins;
103  fState->SetMode(MODE_STRATIFIED);
104  }
105  }
106 
107  {
108  size_t tot_boxes = pow( boxes, N);
109  fState->SetCallsPerBox(std::max( this->GetNCalls() / tot_boxes, (size_t)2) );
110  this->SetNCalls( fState->GetCallsPerBox() * tot_boxes);
111  }
112 
113  /* total volume of x-space/(avg num of calls/bin) */
114  fState->SetJacobian( fState->GetVolume() * pow((GReal_t) bins, (GReal_t)N)/ GetNCalls());
115 
116  fState->SetNBoxes(boxes);
117 
118  /* If the number of bins changes from the previous invocation, bins
119  are expanded or contracted accordingly, while preserving bin
120  density */
121 
122  if (bins != fState->GetNBins()) {
123  ResizeGrid( bins);
124 
125  if (fState->GetVerbose() > 1) {
126  PrintGrid();
127  }
128  }
129 
130  if (fState->GetVerbose() >= 0) {
131  PrintHead();
132  }
133  }
134 
135  fState->SetItStart( fState->GetItNum());
136 
137  cum_int = 0.0;
138  cum_sig = 0.0;
139 
140  size_t calls_per_box = fState->GetCallsPerBox();
141  GReal_t jacbin = fState->GetJacobian();
142  //GReal_t *x = fState->GetX().data();
143  //GInt_t *bin = fState->GetBin().data();
144 
145  //fState->AllocateResources();
146 
147 
148  for (size_t it = 0; it < fState->GetIterations(); it++) {
149 
150  auto start = std::chrono::high_resolution_clock::now();
151 
152  GReal_t intgrl = 0.0;
153  GReal_t intgrl_sq = 0.0;
154  GReal_t tss = 0.0;
155  GReal_t wgt, var, sig;
156 
157 
158  fState->SetItNum(fState->GetItStart() + it);
159 
160  ResetGridValues();
161 
162  /*
163  * **********************************************
164  * call accelerator *
165  * **********************************************
166  */
167  ProcessFuncionCalls( fFunctor, intgrl, tss);
168  /*
169  * Compute final results for this iteration
170  */
171 
172  var = tss / (calls_per_box - 1.0);
173 
174 
175  if (var > 0) {
176  wgt = 1.0 / var;
177  } else if (fState->GetSumOfWeights() > 0) {
178  wgt = fState->GetSumOfWeights() / fState->GetSamples();
179  } else {
180  wgt = 0.0;
181  }
182 
183  intgrl_sq = intgrl * intgrl;
184 
185  sig = sqrt(var);
186 
187 
188  if (wgt > 0.0) {
189 
190  GReal_t sum_wgts = fState->GetSumOfWeights();
191  GReal_t wtd_int_sum = fState->GetWeightedIntSum();
192  GReal_t m = (sum_wgts > 0) ? (wtd_int_sum / sum_wgts) : 0;
193  GReal_t q = intgrl - m;
194 
195  fState->SetSamples(fState->GetSamples()+1);
196  fState->SetSumOfWeights(fState->GetSumOfWeights() + wgt);
197  fState->SetWeightedIntSum(fState->GetWeightedIntSum() + intgrl * wgt);
198  fState->SetChiSum(fState->GetChiSum() + intgrl_sq * wgt );
199 
200  cum_int = fState->GetWeightedIntSum() / fState->GetSumOfWeights();
201  cum_sig = sqrt(1.0/fState->GetSumOfWeights() );
202 
203 #if USE_ORIGINAL_CHISQ_FORMULA
204 
205  /*
206  * This is the chisq formula from the original Lepage paper. It
207  * computes the variance from <x^2> - <x>^2 and can suffer from
208  * catastrophic cancellations, e.g. returning negative chisq.
209  */
210 
211  if (fState->GetSamples() > 1)
212  {
213  fState->SetChiSquare((fState->GetChiSum() - fState->GetWeightedIntSum() * cum_int)/(fState->GetSamples() - 1.0));
214  }
215 #else
216  /*
217  * The new formula below computes exactly the same quantity as above
218  * but using a stable recurrence
219  */
220 
221  if (fState->GetSamples() == 1) {
222  fState->SetChiSquare(0.0);
223  } else {
224 
225  GReal_t chi2 = fState->GetChiSquare();
226 
227  chi2 *= (fState->GetSamples() - 2.0);
228  chi2 += (wgt / (1 + (wgt / sum_wgts))) * q * q;
229  chi2 /= (fState->GetSamples() - 1.0);
230 
231  fState->SetChiSquare(chi2);
232 
233  }
234 #endif
235  } else {
236  cum_int += (intgrl - cum_int) / (it + 1.0);
237  cum_sig = 0.0;
238  }
239 
240  auto end = std::chrono::high_resolution_clock::now();
241  auto elapsed = std::chrono::duration_cast<std::chrono::microseconds>(end - start);
242 
243  fState->SetResult(intgrl);
244  fState->SetSigma(sig);
245 
246 
247  fState->StoreIterationResult(intgrl, sig);
248  fState->StoreCumulatedResult(cum_int, cum_sig);
249  fState->StoreIterationDuration( GReal_t(elapsed.count())/1000 );
250 
251 
252  if (fState->GetVerbose() >= 0) {
253  PrintResults( intgrl, sig, cum_int, cum_sig, GReal_t(elapsed.count())/1000);
254 
255  if (it + 1 == fState->GetIterations() && fState->GetVerbose() > 0) {
256  PrintGrid();
257  }
258  }
259 
260  if (fState->GetVerbose() > 1) {
261  PrintDistribution();
262  }
263 
264  RefineGrid();
265 
266  if (fState->GetVerbose() > 1) {
267  PrintGrid();
268  }
269 
270  if( (fState->IsUseRelativeError()) && (cum_sig/cum_int < fState->GetMaxError()) ) break;
271  if( it > 1 && (!fState->IsUseRelativeError()) && (fabs(cum_int - intgrl )/cum_int < fState->GetMaxError()) ) break;
272 
273  }
274 
275  /* By setting stage to 1 further calls will generate independent
276  estimates based on the same grid, although it may be rebinned. */
277 
278  if(reset){
279  fState->SetStage(0);
280  }
281  else fState->SetStage(1);
282 
283  fResult=cum_int;
284  fAbsError=cum_sig;
285 
286  return 0;
287

411 template<size_t N, typename GRND >
412 void Vegas< N , GRND>::ResizeGrid(const GInt_t bins) {
413 
414 
415 
416  /* weight is ratio of bin sizes */
417 
418  GReal_t pts_per_bin = (GReal_t) fState->GetNBins() / (GReal_t) bins;
419 
420  for (size_t j = 0; j < N; j++) {
421  GReal_t xold;
422  GReal_t xnew = 0;
423  GReal_t dw = 0;
424  GInt_t i = 1;
425 
426  for ( size_t k = 1; k <= fState->GetNBins(); k++) {
427  dw += 1.0;
428  xold = xnew;
429  xnew = GetCoordinate(k, j);
430 
431  for (; dw > pts_per_bin; i++) {
432  dw -= pts_per_bin;
433  SetNewCoordinate(i, xnew - (xnew - xold) * dw);
434  }
435  }
436 
437  for ( size_t k = 1; k < bins; k++) {
438  SetCoordinate(k, j, GetNewCoordinate(k));
439  }
440 
441  SetCoordinate( bins, j, 1);
442  }
443 
444  fState->SetNBins(bins);

448 template<size_t N , typename GRND>
449 void Vegas< N, GRND >::RefineGrid() {
450 
451 
452  for (size_t j = 0; j < N; j++) {
453  GReal_t grid_tot_j, tot_weight;
454  //GReal_t *weight = fState->GetWeight();
455 
456  GReal_t oldg = GetDistributionValue(0, j);
457  GReal_t newg = GetDistributionValue(1, j);
458 
459  SetDistributionValue(0, j, (oldg + newg) / 2);
460  grid_tot_j = GetDistributionValue(0, j);
461 
462 
463  for (size_t i = 1; i < fState->GetNBins() - 1; i++) {
464 
465  GReal_t rc = oldg + newg;
466 
467  oldg = newg;
468 
469  newg = GetDistributionValue(i + 1, j);
470 
471  SetDistributionValue( i, j, (rc + newg) / 3);
472 
473  grid_tot_j += GetDistributionValue(i, j);
474  }
475 
476  SetDistributionValue( fState->GetNBins() - 1, j, (newg + oldg) / 2);
477 
478  grid_tot_j += GetDistributionValue(fState->GetNBins() - 1, j);
479 
480  tot_weight = 0;
481 
482  for (size_t i = 0; i < fState->GetNBins(); i++) {
483  fState->SetWeight(i,0);
484 
485  if (GetDistributionValue( i, j)> 0)
486  {
487  oldg = grid_tot_j / GetDistributionValue( i, j);
488  /* damped change */
489  fState->SetWeight(i,pow (((oldg - 1) / oldg / log (oldg)), fState->GetAlpha() ));
490  }
491 
492  tot_weight += fState->GetWeight()[i];
493 
494  #ifdef DEBUG
495  fState->GetOStream() << ("weight[%d] = %g\n") % i % fState->GetWeight()[i] ;
496  #endif
497  }
498 
499  {
500  GReal_t pts_per_bin = tot_weight / fState->GetNBins();
501 
502  GReal_t xold;
503  GReal_t xnew = 0;
504  GReal_t dw = 0;
505  size_t i = 1;
506 
507  for (size_t k = 0; k < fState->GetNBins(); k++) {
508  dw += fState->GetWeight()[k];
509  xold = xnew;
510  xnew = GetCoordinate( k + 1, j);
511 
512  for (; dw > pts_per_bin; i++) {
513  dw -= pts_per_bin;
514  SetNewCoordinate( i, xnew - (xnew - xold) * dw / fState->GetWeight()[k] );
515  }
516  }
517 
518  for (size_t k = 1; k < fState->GetNBins(); k++) {
519 
520  SetCoordinate(k, j, GetNewCoordinate(k));
521  }
522 
523  SetCoordinate(fState->GetNBins(), j, 1.0);
524  }
525  }
