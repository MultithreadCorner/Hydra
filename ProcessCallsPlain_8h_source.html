50 // ProcessCallsPlainUnary is a functor that takes in a value x and

82  __host__ __device__ inline
83  PlainState operator()(size_t index)
84  {
85 
86  GRND randEng(fSeed);
87  randEng.discard(index);
88  HYDRA_EXTERNAL_NS::thrust::uniform_real_distribution<GReal_t> uniDist(0.0, 1.0);
89 
90  GReal_t x[N];
91 
92  for (size_t j = 0; j < N; j++) {
93  GReal_t r = uniDist(randEng);
94  x[j] = fXLow[j] + r*fDeltaX[j];

96

104  result.fM2 = 0;

117 // ProcessCallsPlainBinary is a functor that accepts two PlainState
118 // structs and returns a new summary_stats_data which are an
119 // approximation to the summary_stats for
120 // all values that have been agregated so far
121 struct ProcessCallsPlainBinary
122  : public HYDRA_EXTERNAL_NS::thrust::binary_function<PlainState const&,
123  PlainState const&,
124  PlainState >
125 {
126  __host__ __device__ inline
127  PlainState operator()(const PlainState& x, const PlainState& y)
128  {
129  PlainState result;
130 
131  // precompute some common subexpressions
132  size_t n = x.fN + y.fN;
133 
134  GReal_t delta = y.fMean - x.fMean;
135  GReal_t delta2 = delta * delta;
136 
137  //Basic number of samples (n), min, and max
138  result.fN = n;
139  result.fMin = HYDRA_EXTERNAL_NS::thrust::min(x.fMin, y.fMin);
140  result.fMax = HYDRA_EXTERNAL_NS::thrust::max(x.fMax, y.fMax);
141 
142  result.fMean = (x.fMean* x.fN + y.fMean* y.fN) / n;
143 
144  result.fM2 = x.fM2 + y.fM2;
145  result.fM2 += delta2 * x.fN * y.fN / n;
146 
147  return result;
148  }
149 };
